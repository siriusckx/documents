## IO多路复用实践
### 用select函数同时读取鼠标和键盘
```
int ret = -1;
fd_set myset;

//当前有2个fd，一个是fd一个是0
//处理myset
FD_ZERO(&myset);
FD_SET(fd, &myset);
FD_SET(0, &myset);

tm.tv_sec = 10;
tm.tv_usec = 0;

ret = select(fd+1, &myset, NULL, NULL, &tm);
if(ret < 0)
{
    //错误
    return -1;
}else if(ret == 0)
{
    //超时
}else
{
    //等到一个事件
    if(FD_ISSET(fd, &myset))
    {
        //处理鼠标事件
        memset(buf, 0 , sizeof(buf));
        ret = read(fd, buf ,500);
        if(ret > 0)
        {
            printf("鼠标读出的内容是：[%s].\n", buf);
        }
    }

    if(FD_ISSET(0, &myset))
    {
        //处理键盘事件
        memset(buf, 0, sizeof(buf));
        read(0, buf, 5);
        if(ret > 0)
        {
            printf("键盘读出的内容是：[%s].\n", buf);
        }
    }
}
```
### 用poll函数同时读取鼠标和键盘
```
struct pollfd myfds[2];

//初始化myfds
myfds[0].fd = 0;             //键盘
myfds[0].events = POLLIN;    //等待读操作

myfds[0].fd = fd;             //鼠标
myfds[0].events = POLLIN;    //等待读操作
ret =  poll(myfds, fd+1, 10000);
if(ret < 0 )
{
    //错误
    return -1;
}else if(ret == 0)
{
    //超时
}
else
{
    //捕捉到事件
    if(myfds[0].events == myfds[0].revents)
    {
        //读键盘数据
    }
    if(myfds[1].events == myfds[1].revents)
    {
        //读鼠标数据
    }
}
```
