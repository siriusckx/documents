## while循环
```
num = 3

while [ $num -gt 0 ]
do
    echo $num
    num = $(($num-1))
    sleep 1
done
    echo $num
```

```
#创建10个用户，并为其设置初始密码
i = 0
while [ $i -lt 10]
do
    echo '123'|passwd --stdin user$i
    i = $(($i+1))
done
```

```
#隔行输出不同颜色内容
i = 0

whle [ $i -lt 10]
do
    if[ $(($i%2)) -eq 0 ];then
        echo -e "\033[37;40m${i}\033[0m"
    else
        echo $i
    fi
    i = $(($i+1))
done
```

## for循环
> shell里面只支持for in
```
#从文件导入用户，并创建用户 
for i in `cat user.txt`
do
    echo $i
    useradd $i
    echo 123|passwd --stdin $i
done
```
```
#将passwd下的用户名提取出来，在users目录下创建对应的目录
for i in `cat /etc/passwd|awk -F: '{print $1}'`
do
    echo $i
    mkdir -p users/$i
done
```
